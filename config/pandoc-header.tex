% Font fallback configuration for Unicode symbols
\directlua{
  luaotfload.add_fallback("notofallback", {
    "NotoSansMath:mode=harf;",
    "NotoSansSymbols:mode=harf;",
    "NotoSansSymbols2:mode=harf;",
    "NotoColorEmoji:mode=harf;",
  })
}

\setmainfont{Noto Sans}[RawFeature={fallback=notofallback}]
\setsansfont{Noto Sans}[RawFeature={fallback=notofallback}]
\setmonofont{Noto Sans Mono}[RawFeature={fallback=notofallback}]

% Compact tables: 8pt font + tighter column padding to prevent overlap
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\footnotesize\setlength{\tabcolsep}{3pt}}
